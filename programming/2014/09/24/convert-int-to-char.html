Конвертирование Char в Int и наоборот

Sep 24, 2014

Пример кода чтоли

#include <conio.h>
#include <stdio.h>
#include <stdlib.h>

int main(void)
{
    char ch;

    do {
        ch = getch();

        if(islower(ch))
            ch = toupper(ch);
        else
            ch = tolower(ch);
        putchar(ch);
    } while (ch!='.');

    return 0;
}
